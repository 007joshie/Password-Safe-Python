def Encrypt(txt,n):
    result = ''
    for i in txt:
        c = ord(i)+n
        if ord(i) != c:
            b = chr(c)
            result += b
    return result

def Decrypt(txt,n):
    decrypt = ''    
    for x in txt:
        c = ord(x)-n
        if ord(x) != c:
            b = chr(c)
            decrypt += b
    return decrypt
